# My first folium map

```python
import folium

width, height = 650, 500
radars = folium.Map(location=[40, -122], zoom_start=5,
                    tiles='OpenStreetMap',
                    width=width, height=height)

for name, location in locations.items():
    radars.simple_marker(location=location, popup=name)
```
